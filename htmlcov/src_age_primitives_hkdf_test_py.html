from .hkdf import hkdf 

 

 

def test_hkdf(): 

"""Test Case 1, RFC 5869, section A.1""" 

 

keying_material = bytes.fromhex("0b0b0b0b0b0b0b0b0b0b0b0b0b0b0b0b0b0b0b0b0b0b") 

assert len(keying_material) == 22 

 

salt = bytes.fromhex("000102030405060708090a0b0c") 

assert len(salt) == 13 

 

info = bytes.fromhex("f0f1f2f3f4f5f6f7f8f9") 

assert len(info) == 10 

 

L = 42 

 

key = hkdf(salt, info, keying_material, L) 

assert len(key) == L 

 

assert key == bytes.fromhex( 

"3cb25f25faacd57a90434f64d0362f2a2d2d0a90cf1a5a4c5db02d56ecc4c5bf" + "34007208d5b887185865" 

)
